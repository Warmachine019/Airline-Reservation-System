CREATE DATABASE airline_reservation;
USE airline;

CREATE TABLE flights (
    id INT AUTO_INCREMENT PRIMARY KEY,
    depart_from VARCHAR(50),
    destination VARCHAR(50),
    price DECIMAL(10,2)
);
